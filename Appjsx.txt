// Importar todas las pantallas
import HomeScreen from './HomeScreen';
import EntradaScreen from './EntradaScreen';
import SalidaScreen from './SalidaScreen';
import SettingsScreen from './SettingsScreen';

export default function App() {
  const [page, setPage] = useState('home');

  return (
    <View style={styles.container}>
      {/* PANTALLAS PRINCIPALES */}
      {page === 'home' && <HomeScreen onNavigate={setPage} />}
      {page === 'entrada' && <EntradaScreen onNavigate={setPage} />}
      {page === 'salida' && <SalidaScreen onNavigate={setPage} />}

      {/* PANTALLAS SECUNDARIAS (FASE 3) */}
      {page === 'Configuranza' && <SettingsScreen onNavigate={setPage} />}
      {page === 'analytics' && <AnalyticsPlaceholder onNavigate={setPage} />}
      {page === 'alertas' && <AlertasPlaceholder onNavigate={setPage} />}

      {/* PANTALLAS DE DASHBOARD ITEMS */}
      {page === 'inventario' && <InventarioPlaceholder onNavigate={setPage} />}
      {page === 'sin-stock' && <SinStockPlaceholder onNavigate={setPage} />}

      {/* FOOTER NAVBAR - SOLO EN PANTALLAS PRINCIPALES */}
      {(page === 'home' || page === 'entrada' || page === 'salida') && (
        <View style={styles.navbar}>
          {/* Botón Dashboard */}
          <TouchableOpacity
            style={[
              styles.navBtn,
              page === 'home' && styles.navBtnActive,
            ]}
            onPress={() => setPage('home')}
          >
            <Text style={styles.navIcon}>🏠</Text>
            <Text style={styles.navLabel}>Dashboard</Text>
          </TouchableOpacity>

          {/* Botón ADD (Entrada) */}
          <TouchableOpacity
            style={[
              styles.navBtn,
              page === 'entrada' && styles.navBtnActive,
            ]}
            onPress={() => setPage('entrada')}
          >
            <Text style={styles.navIcon}>➕</Text>
            <Text style={styles.navLabel}>ADD</Text>
          </TouchableOpacity>

          {/* Botón VENTA (Salida) */}
          <TouchableOpacity
            style={[
              styles.navBtn,
              page === 'salida' && styles.navBtnActive,
            ]}
            onPress={() => setPage('salida')}
          >
            <Text style={styles.navIcon}>💰</Text>
            <Text style={styles.navLabel}>VENTA</Text>
          </TouchableOpacity>
        </View>
      )}
    </View>
  );
}
